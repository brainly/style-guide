## Examples

### Fill Mode

<Story story={stories.FillMode} />

```jsx
const fillModes = ['none', 'forwards', 'backwards', 'both'];

const shrinkFadeEffect = {
  initial: {
    opacity: 0,
  },
  animate: {
    opacity: 1,
    transform: {scale: 0.25},
    duration: 1000,
    easing: 'entry',
  },
  exit: {
    opacity: 0,
    duration: 1000,
    easing: 'exit',
  },
};

fillModes.map(mode => (
  <Container key={mode} description={mode}>
    <Transition
      active={active}
      effect={shrinkFadeEffect}
      fillMode={mode}
      delay={1000}
    >
      <Circle />
    </Transition>
  </Container>
));
```
