<Canvas>
  <Story name="States Dark">
    {args => (
      <StoryVariantTable>
        <thead>
          <tr>
            <th />
            <th>
              <Text weight="bold" transform="capitalize" size="medium">
                default
              </Text>
            </th>
            <th>

                error
              </Text>
            </th>
          </tr>
        </thead>
        <tbody>
          {['outline', 'solid'].map(variant => {
            return [false, true].map(indicator => {
              return [false, true].map(checked => {

                        variant={variant}

                        value="default"
                      >

                          <CardRadioGroup.Indicator slot="top-left" />
                        )}
                        <Flex
                          justifyContent="center"
                          alignItems="center"
                          fullHeight
                        >
                          <Text
                            color="text-gray-50"
                            size="xsmall"
                            weight="bold"

                            Placeholder
                          </Text>
                        </Flex>
                      </CardRadioGroup.Item>
                    </CardRadioGroup>
                  </td>
                ));
                return (
                  <tr>
                    <td>
                      <Text weight="bold" size="small" transform="capitalize">
                        {variant}

                        {checked ? 'checked' : 'unchecked'}
                        <br />

                          ? 'with indicator'
                          : 'without indicator'}{' '}

                    </td>

              });
