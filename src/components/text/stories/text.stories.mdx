export const getValues = function(object, addUndefined = true) {
  return addUndefined
    ? [undefined, ...Object.values(object)]
    : Object.values(object);
};

# Typography

The Brainly brand typeface is Proxima Nova, a geometric sans serif that works equally well with large headlines and functional copy. Use this typeface according to the following rules for consistency across the brand.

Proxima Nova combines modern proportions with a geometric appearance and has become one of the most popular web fonts, in use on thousands of websites around the world.

export const HeadlineExamples = (extraBold) => {
  const sizeVariant = [];
  getValues(HEADLINE_SIZE, false)
    .reverse()
    .forEach((size) => {
      let itemSize;
      textSizesMap.map((item) =>
        item.type === size ? (itemSize = `${item.fontSize}`) : null,
      );
      sizeVariant.push(
        <TypeSpecimen key={size}>
          <Label>{size}</Label>
          <Headline
            key={size}
            type={HEADLINE_TYPE.H2}
            size={size}
            extraBold={extraBold}
          >
            Headline
          </Headline>
        </TypeSpecimen>,
      );
    });
  return sizeVariant;
};
