## Stories

### Animation on Hover

<Canvas>
  <Story name="Hover">
    {args => {
      // active animation on button hover
      const [active, setActive] = React.useState(false);
      return (
        <div style={{width: 300}}>
          <Sparks
            {...args}
            variant="l"
            shape="spark"
            duration={2000}
            delay={600}
            iterationCount={5}
            active={active}
          >
            <Button
              variant="solid-indigo"
              size="l"
              onMouseEnter={() => setActive(true)}
              onMouseLeave={() => setActive(false)}
            >
              Button
            </Button>
          </Sparks>
        </div>
      );
    }}
  </Story>
</Canvas>
