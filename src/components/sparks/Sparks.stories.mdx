### Animation on Hover

<Canvas>
  <Story name="Hover">
    {args => {
      // active animation on button hover
      const [active, setActive] = React.useState(false);
      return (
        <div style={{width: 300}}>
          <Sparks {...args} variant="l" active={active}>

            </Button>
