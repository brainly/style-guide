### Borders

<Canvas>
  <Story name="Borders">
    {args => (
      <Flex direction="column">
        {Object.values(COLOR).map(color => (
          <Flex direction="column" key={color} marginBottom="s">
            <Flex marginBottom="xs">{color}</Flex>
            {color === COLOR.white ? (
              <Box noBorderRadius color="indigo-40">
                <Box
                  {...args}
                  color="indigo-20"
                  border
                  borderColor={COLOR[color]}
                />
              </Box>
            ) : (
              <Box {...args} color="indigo-10" border borderColor={color} />
            )}
          </Flex>
        ))}
      </Flex>
    )}
  </Story>
</Canvas>
