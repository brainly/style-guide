### States

<Canvas>
  <Story name="States">
    {args => (
      <StoryVariantTable>
        <thead>
          <tr>
            <th />
            <th>
              <Text weight="bold" transform="capitalize" size="medium">
                default
              </Text>
            </th>
            <th>
              <Text weight="bold" transform="capitalize" size="medium">
                disabled
              </Text>
            </th>
            <th>
              <Text weight="bold" transform="capitalize" size="medium">
                error
              </Text>
            </th>
          </tr>
        </thead>
        <tbody>
          {['outline', 'solid'].map(variant => {
            return [false, true].map(indicator => {
              return [false, true].map(checked => {
                const states = ['default', 'disabled', 'error'].map(states => (
                  <td>
                    <CardRadio
                      {...args}
                      variant={variant}
                      width="120px"
                      height="64px"
                      disabled={states === 'disabled'}
                      invalid={states === 'error'}
                      defaultChecked={checked}
                    >
                      {indicator && <CardRadio.Indicator slot="top-left" />}
                      <Flex
                        justifyContent="center"
                        alignItems="center"
                        fullHeight
                      >
                        <Text color="text-gray-50" size="xsmall" weight="bold">
                          Placeholder
                        </Text>
                      </Flex>
                    </CardRadio>
                  </td>
                ));
                return (
                  <tr>
                    <td>
                      <Text weight="bold" size="small" transform="capitalize">
                        {variant}
                        <br />
                        {checked ? 'checked' : 'unchecked'}
                        <br />
                        {indicator
                          ? 'with indicator'
                          : 'without indicator'}{' '}
                        <br />
                      </Text>
                    </td>
                    {states}
                  </tr>
                );
              });
            });
          })}
        </tbody>
      </StoryVariantTable>
    )}
  </Story>
</Canvas>
